INICIO
	LER(horaSaida, minSaida, horaDuraçao, minDuraçao)

	auxDuraçao = (horaDuraçao*60 + minDuraçao
	auxChegada = auxSaida + auxDuraçao
	SE (auxChegada > 1440)
		Escrever("Chega no dia seguinte")
		auxChegada = auxChegada - 1440
	SENAO 
		ESCREVER("Não chega no dia seguinte") 
	FIMSE
	horaChegada = auxChegada DIV 60
	minChegada = auxChegada MOD 60
	ESCREVER("Hora de chegada",horaChegada,":",minChegada)

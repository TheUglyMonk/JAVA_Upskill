	LER(num)

 		ESCREVER(“Número não tem 3 dígitos”)

	dig2 = (num DIV 10) MOD 10 
	dig1 = (num DIV 100) MOD 10 
	ESCREVER(dig1, “ ”, dig2, “ ”, dig3) 
	FIMSE

 		ESCREVER(“Impar”) 
	SENAO
